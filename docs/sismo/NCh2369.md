# NCh2369:2025 — Espectro Sísmico de Diseño

Genera y visualiza el espectro de respuesta de diseño según la norma chilena **NCh2369:2025** para estructuras industriales.

---

## Fundamentos

### Forma espectral

La aceleración espectral de diseño $S_a(T)$ se obtiene en base al espectro de referencia y factores $R^*$, I y $\eta$:

**1. Espectro de Referencia §5.4.2** (sin reducción):

El espectro de referencia tiene por objetivo representar un nivel razonable de la amenaza sísmica esperada en las distintas zonas sísmicas, a diferencia del espectro de diseño; ha sido definido para la estimación de desplazamientos y para el desarrollo de los análisis especiales. Se estima que el espectro de referencia representa un nivel de amenaza uniforme razonable y consistente con un periodo de retorno medio 475 años para una razón de amortiguamiento de $5\%$.

$$S_{aH}(T_H) = A_r \cdot S \cdot \frac{1 + r \cdot \left(\dfrac{T_H}{T_0}\right)^p}{1 + \left(\dfrac{T_H}{T_0}\right)^q}$$

$$S_{aV}(T_V) = 0.7 \cdot A_r \cdot S \cdot \frac{1 + r \cdot \left(1.7 \dfrac{T_V}{T_0}\right)^p}{1 + \left(1.7 \dfrac{T_V}{T_0}\right)^q}$$

La razón de amortiguamiento incorporada en los espectros de referencia es $\xi = 0,05$. Para razones de amortiguamiento menores, se debe ponderar 
el espectro que corresponda por el factor $(0,05/\xi)^{0,4}$. Esta expresión es válida sólo para valores de $\xi$ entre 0,02 y 0,05.
El uso de razones de amortiguamiento mayores que 0,03 en la dirección vertical se debe fundamentar mediante un análisis racional.
Los espectros de referencia para la dirección horizontal y vertical corresponden a **demandas asociadas a nivel último (LRFD)**.


**2. Espectro de diseño §5.4.1** (reducido):

$$S_a(T_H) = \frac{I \cdot \eta}{R^*} \cdot S_{aH}(T_H)$$

$$S_a(T_V) = \frac{I \cdot \eta}{R_V} \cdot S_{aV}(T_V)$$

**2.1 Factor de Importancia $I$ (§ 4.3.2)**

| Categoría | # |
| | |
| I | 0.80 |
| II | 1.00 |
| III | 1.20 |
| IV | 1.20 |

**2.2 Factor de amoritguamiento $\eta$**

$$\eta = \left(\frac{0.05}{\xi_{H/V}}\right)^{0.4}$$

**2.3 Factor de Modificación de Respuesta $R^*$**

Es relativo al periodo del modo con mayor masa de traslación equivalente, en la dirección de análisis.

$$R^* = \begin{cases} 1 & R = 1 \\ R & R \neq 1 \quad T^* \geq C_r T_1 \\ 1.5 + (R - 1.5) \dfrac{T^*}{C_r T_1} & R \neq 1 \quad T^* < C_r T_1 \end{cases}$$

$$C_r = 0.16 \cdot R$$

Donde $R$ es el factor de modificación de respuesta **horizontal** relativo a la configuración estructural que se considera (§Tabla 7) y $T^*$ corresponde al periodo fundamental en la dirección horizontal de análisis sísmico, el cual generalmente se evalúa en el rango de periodos espectrales (0-5 seg).

<script>
(function () {
  const AR_BY_ZONE = { 1: 0.28, 2: 0.42, 3: 0.56 };
  const SOIL_PARAMS = {
    "A": { S: 0.90, r: 4.50, T0: 0.15, p: 1.85, q: 3.00, T1: 0.15 },
    "B": { S: 1.00, r: 4.50, T0: 0.30, p: 1.60, q: 3.00, T1: 0.27 },
    "C": { S: 1.05, r: 4.50, T0: 0.40, p: 1.50, q: 3.00, T1: 0.35 },
    "D": { S: 1.00, r: 3.50, T0: 0.60, p: 1.00, q: 2.50, T1: 0.41 },
    "E": { S: 1.00, r: 3.00, T0: 1.20, p: 1.00, q: 2.70, T1: 0.79 },
  };

  function specShape(ar, sp, T, scaleFactor, periodShift) {
    if (T === 0) return scaleFactor * ar * sp.S;
    const ratio = (periodShift * T) / sp.T0;
    return scaleFactor * ar * sp.S *
      (1 + sp.r * Math.pow(ratio, sp.p)) /
      (1 + Math.pow(ratio, sp.q));
  }

  function rStar(T, R, t1) {
    const limit = 0.16 * R * t1;
    if (limit <= 0 || T >= limit) return R;
    return 1.5 + (R - 1.5) * (T / limit);
  }

  function computeSpectrum(zone, soil, I, R, damp, scaleFactor, periodShift, applyRStar) {
    const ar = AR_BY_ZONE[zone];
    const sp = SOIL_PARAMS[soil];
    const eta = Math.pow(0.05 / damp, 0.4);
    const periods = [], accels = [];
    for (let i = 0; i <= 500; i++) {
      const T = Math.round(i * 0.01 * 1e4) / 1e4;
      const sa = specShape(ar, sp, T, scaleFactor, periodShift);
      const rEff = applyRStar ? rStar(T, R, sp.T1) : R;
      periods.push(T);
      accels.push(I * eta * sa / rEff);
    }
    return { periods, accels };
  }

  function interp(accels, T) {
    return accels[Math.min(Math.round(T / 0.01), accels.length - 1)];
  }

  window.calcSpectrum = function () {
    if (typeof Plotly === 'undefined') {
      alert('Plotly aún está cargando, intenta en un momento.');
      return;
    }
    const zone  = parseInt(document.getElementById('zona').value);
    const soil  = document.getElementById('suelo').value;
    const I     = parseFloat(document.getElementById('specI').value);
    const Rx    = parseFloat(document.getElementById('Rx').value);
    const xix   = parseFloat(document.getElementById('xix').value) / 100;
    const Ry    = parseFloat(document.getElementById('Ry').value);
    const xiy   = parseFloat(document.getElementById('xiy').value) / 100;
    const Rv    = parseFloat(document.getElementById('Rv').value);
    const xiv   = parseFloat(document.getElementById('xiv').value) / 100;

    const sp    = SOIL_PARAMS[soil];
    const specX = computeSpectrum(zone, soil, I, Rx, xix, 1.0, 1.0, true);
    const specV = computeSpectrum(zone, soil, I, Rv, xiv, 0.7, 1.7, false);

    const isDark = document.body.getAttribute('data-md-color-scheme') === 'slate';
    const tmpl   = isDark ? 'plotly_dark' : 'plotly_white';

    const traces = [{
      x: specX.periods, y: specX.accels, mode: 'lines',
      name: 'Horizontal X  (R=' + Rx + ', ξ=' + (xix * 100).toFixed(1) + '%)',
      line: { color: '#1f77b4', width: 2.5 }
    }];

    let specY = specX;
    if (Ry !== Rx || xiy !== xix) {
      specY = computeSpectrum(zone, soil, I, Ry, xiy, 1.0, 1.0, true);
      traces.push({
        x: specY.periods, y: specY.accels, mode: 'lines',
        name: 'Horizontal Y  (R=' + Ry + ', ξ=' + (xiy * 100).toFixed(1) + '%)',
        line: { color: '#ff7f0e', width: 2.5, dash: 'dash' }
      });
    }

    traces.push({
      x: specV.periods, y: specV.accels, mode: 'lines',
      name: 'Vertical  (R=' + Rv + ', ξ=' + (xiv * 100).toFixed(1) + '%)',
      line: { color: '#2ca02c', width: 2, dash: 'dot' }
    });

    const saMax = Math.max.apply(null, specX.accels);
    traces.push({
      x: [sp.T0, sp.T0], y: [0, saMax * 1.08], mode: 'lines',
      name: 'T₀ = ' + sp.T0 + ' s (Suelo ' + soil + ')',
      line: { color: 'grey', width: 1, dash: 'dashdot' }
    });

    Plotly.react('spectrum-plot', traces, {
      template: tmpl,
      title: {
        text: 'Espectro NCh2369:2025 — Zona ' + zone + ', Suelo ' + soil + ', I = ' + I,
        font: { size: 14 }
      },
      xaxis: { title: 'Período T [s]', range: [0, 5], dtick: 0.5, minor: { dtick: 0.1 } },
      yaxis: { title: 'Sa [g]', rangemode: 'tozero' },
      legend: { x: 0.65, y: 0.98, bgcolor: 'rgba(0,0,0,0)' },
      margin: { t: 50, b: 60, l: 65, r: 20 },
      hovermode: 'x unified'
    }, { responsive: true });

    // ── Tabla de valores ─────────────────────────────────────────────────────
    const T_TABLA = [0, 0.05, 0.10, 0.15, 0.20, 0.30, 0.40, 0.50,
                     0.75, 1.00, 1.25, 1.50, 2.00, 2.50, 3.00, 4.00, 5.00];
    const showY = (Ry !== Rx || xiy !== xix);
    let html = '<table><thead><tr><th>T [s]</th><th>Sa X [g]</th>';
    if (showY) html += '<th>Sa Y [g]</th>';
    html += '<th>Sa V [g]</th></tr></thead><tbody>';
    for (const T of T_TABLA) {
      html += '<tr><td>' + T.toFixed(2) + '</td>';
      html += '<td>' + interp(specX.accels, T).toFixed(4) + '</td>';
      if (showY) html += '<td>' + interp(specY.accels, T).toFixed(4) + '</td>';
      html += '<td>' + interp(specV.accels, T).toFixed(4) + '</td></tr>';
    }
    html += '</tbody></table>';
    document.getElementById('spectrum-table').innerHTML = html;
  };
})();
</script>
